146 [[nodiscard]] async_simple::coro::Lazy<std::errc> connect(
147 std::string host, std::string port,
148 std::chrono::steady_clock::duration timeout_duration =
149 std::chrono::seconds(5)) {
150#ifdef ENABLE_SSL
151 if (!ssl_init_ret_) {
152 co_return std::errc::not_connected;
153 }
154#endif
155 if (has_closed_) [[unlikely]] {
156 ELOGV(ERROR,
157 "a closed client is not allowed connect again, please create a new "
158 "client");
159 co_return std::errc::io_error;
160 }
161
162 ELOGV(INFO, "client_id %d begin to connect %s", client_id_, port.data());
163 async_simple::Promise<async_simple::Unit> promise;
164 asio_util::period_timer timer(get_io_context());
165 timeout(timer, timeout_duration, promise, "connect timer canceled")
166 .via(&executor_)
167 .detach();
168
169 std::error_code ec = co_await asio_util::async_connect(get_io_context(),
170 socket_, host, port);
171 std::error_code err_code;
172 timer.cancel(err_code);
173
174 co_await promise.getFuture();
175 if (ec) {
176 if (is_timeout_) {
177 co_return std::errc::timed_out;
178 }
179 co_return std::errc::not_connected;
180 }
181
182 if (is_timeout_) {
183 ELOGV(WARN, "client_id %d connect timeout", client_id_);
184 co_return std::errc::timed_out;
185 }
186
187#ifdef ENABLE_SSL
188 if (use_ssl_) {
189 assert(ssl_stream_);
190 auto shake_ec = co_await asio_util::async_handshake(
191 ssl_stream_, asio::ssl::stream_base::client);
192 if (shake_ec) {
193 ELOGV(WARN, "client_id %d handshake failed: %s", client_id_,
194 shake_ec.message().data());
195 co_return std::errc::not_connected;
196 }
197 }
198#endif
199
200 co_return std::errc{};
201 }
202#ifdef ENABLE_SSL
203 [[nodiscard]] bool init_ssl(const std::string &base_path,
204 const std::string &cert_file,
205 const std::string &domain = "localhost") {
206 try {
207 ssl_init_ret_ = false;
208 ELOGV(INFO, "init ssl: %s", domain.data());
209 auto full_cert_file = std::filesystem::path(base_path).append(cert_file);
210 ELOGV(INFO, "current path %s",
211 std::filesystem::current_path().string().data());
212 if (file_exists(full_cert_file)) {
213 ELOGV(INFO, "load %s", full_cert_file.string().data());
214 ssl_ctx_.load_verify_file(full_cert_file);
215 }
216 else {
217 ELOGV(INFO, "no certificate file %s", full_cert_file.string().data());
218 return ssl_init_ret_;
219 }
220
221 ssl_ctx_.set_verify_mode(asio::ssl::verify_peer);
222
223 // ssl_ctx_.add_certificate_authority(asio::buffer(CA_PEM));
224
225 ssl_ctx_.set_verify_callback(asio::ssl::host_name_verification(domain));
226 ssl_stream_ =
227 std::make_unique<asio::ssl::stream<asio::ip::tcp::socket &>>(
228 socket_, ssl_ctx_);
229 use_ssl_ = true;
230 ssl_init_ret_ = true;
231 } catch (std::exception &e) {
232 ELOGV(ERROR, "init ssl failed: %s", e.what());
233 }
234 return ssl_init_ret_;
235 }

269 template <auto func, typename... Args>
270 async_simple::coro::Lazy<
271 rpc_result<decltype(get_return_type<func>()), coro_rpc_protocol>>
272 call_for(auto duration, Args... args) {
273 using R = decltype(get_return_type<func>());
274
275 if (has_closed_) [[unlikely]] {
276 ELOGV(ERROR,
277 "a closed client is not allowed call again, please create a new "
278 "client");
279 auto ret = rpc_result<R, coro_rpc_protocol>{
280 unexpect_t{},
281 coro_rpc_protocol::rpc_error{
282 std::errc::io_error,
283 "client has been closed, please create a new client"}};
284 co_return ret;
285 }
286
287 rpc_result<R, coro_rpc_protocol> ret;
288#ifdef ENABLE_SSL
289 if (!ssl_init_ret_) {
290 ret = rpc_result<R, coro_rpc_protocol>{
291 unexpect_t{}, coro_rpc_protocol::rpc_error{std::errc::not_connected,
292 "not connected"}};
293 co_return ret;
294 }
295#endif
296
297 static_check<func, Args...>();
298
299 async_simple::Promise<async_simple::Unit> promise;
300 asio_util::period_timer timer(get_io_context());
301 timeout(timer, duration, promise, "rpc call timer canceled")
302 .via(&executor_)
303 .detach();
304
305#ifdef ENABLE_SSL
306 if (use_ssl_) {
307 assert(ssl_stream_);
308 ret = co_await call_impl<func>(*ssl_stream_, std::move(args)...);
309 }
310 else {
311#endif
312 ret = co_await call_impl<func>(socket_, std::move(args)...);
313#ifdef ENABLE_SSL
314 }
315#endif
316
317 std::error_code err_code;
318 timer.cancel(err_code);
319
320 if (is_timeout_) {
321 ret = rpc_result<R, coro_rpc_protocol>{
322 unexpect_t{}, coro_rpc_protocol::rpc_error{std::errc::timed_out,
323 "rpc call timed out"}};
324 }
325
326 co_await promise.getFuture();
327#ifdef UNIT_TEST_INJECT
328 ELOGV(INFO, "client_id %d call %s %s", client_id_,
329 get_func_name<func>().data(), ret ? "ok" : "failed");
330#endif
331 co_return ret;
332 }

342 async_simple::coro::Lazy<bool> timeout(auto &timer, auto duration,
343 auto &promise, std::string err_msg) {
344 timer.expires_after(duration);
345 bool is_timeout = co_await timer.async_await();
346#ifdef UNIT_TEST_INJECT
347 ELOGV(INFO, "client_id %d %s, is_timeout_ %d, %d , duration %d ms",
348 client_id_, err_msg.data(), is_timeout_, is_timeout,
349 std::chrono::duration_cast<std::chrono::milliseconds>(duration)
350 .count());
351#endif
352 if (!is_timeout) {
353 promise.setValue(async_simple::Unit());
354 co_return false;
355 }
356
357 is_timeout_ = is_timeout;
358 close_socket();
359 promise.setValue(async_simple::Unit());
360 co_return true;
361 }

363 template <auto func, typename... Args>
364 void static_check() {
365 using Function = decltype(func);
366 using param_type = function_parameters_t<Function>;
367 if constexpr (!std::is_void_v<param_type>) {
368 using First = std::tuple_element_t<0, param_type>;
369 constexpr bool is_conn = requires { typename First::return_type; };
370
371 if constexpr (std::is_member_function_pointer_v<Function>) {
372 using Self = class_type_t<Function>;
373 if constexpr (is_conn) {
374 static_assert(is_invocable<Function, Self, First, Args...>::value,
375 "called rpc function and arguments are not match");
376 }
377 else {
378 static_assert(is_invocable<Function, Self, Args...>::value,
379 "called rpc function and arguments are not match");
380 }
381 }
382 else {
383 if constexpr (is_conn) {
384 static_assert(is_invocable<Function, First, Args...>::value,
385 "called rpc function and arguments are not match");
386 }
387 else {
388 static_assert(is_invocable<Function, Args...>::value,
389 "called rpc function and arguments are not match");
390 }
391 }
392 }
393 else {
394 if constexpr (std::is_member_function_pointer_v<Function>) {
395 using Self = class_type_t<Function>;
396 static_assert(is_invocable<Function, Self, Args...>::value,
397 "called rpc function and arguments are not match");
398 }
399 else {
400 static_assert(is_invocable<Function, Args...>::value,
401 "called rpc function and arguments are not match");
402 }
403 }
404 }

406 template <auto func, typename Socket, typename... Args>
407 async_simple::coro::Lazy<
408 rpc_result<decltype(get_return_type<func>()), coro_rpc_protocol>>
409 call_impl(Socket &socket, Args... args) {
410 using R = decltype(get_return_type<func>());
411
412 auto buffer = prepare_buffer<func>(std::move(args)...);
413 rpc_result<R, coro_rpc_protocol> r{};
414#ifdef GENERATE_BENCHMARK_DATA
415 std::ofstream file(
416 benchmark_file_path + std::string{get_func_name<func>()} + ".in",
417 std::ofstream::binary | std::ofstream::out);
418 file << std::string_view{(char *)buffer.data(), buffer.size()};
419 file.close();
420#endif
421 std::pair<std::error_code, size_t> ret;
422#ifdef UNIT_TEST_INJECT
423 if (g_action == inject_action::client_send_bad_header) {
424 buffer[0] = (std::byte)(uint8_t(buffer[0]) + 1);
425 }
426 if (g_action == inject_action::client_close_socket_after_send_header) {
427 ret = co_await asio_util::async_write(
428 socket, asio::buffer(buffer.data(), coro_rpc_protocol::REQ_HEAD_LEN));
429 ELOGV(INFO, "client_id %d close socket", client_id_);
430 close();
431 r = rpc_result<R, coro_rpc_protocol>{
432 unexpect_t{}, coro_rpc_protocol::rpc_error{std::errc::io_error,
433 ret.first.message()}};
434 co_return r;
435 }
436 else if (g_action ==
437 inject_action::client_close_socket_after_send_partial_header) {
438 ret = co_await asio_util::async_write(
439 socket,
440 asio::buffer(buffer.data(), coro_rpc_protocol::REQ_HEAD_LEN - 1));
441 ELOGV(INFO, "client_id %d close socket", client_id_);
442 close();
443 r = rpc_result<R, coro_rpc_protocol>{
444 unexpect_t{}, coro_rpc_protocol::rpc_error{std::errc::io_error,
445 ret.first.message()}};
446 co_return r;
447 }
448 else if (g_action ==
449 inject_action::client_shutdown_socket_after_send_header) {
450 ret = co_await asio_util::async_write(
451 socket, asio::buffer(buffer.data(), coro_rpc_protocol::REQ_HEAD_LEN));
452 ELOGV(INFO, "client_id %d shutdown", client_id_);
453 socket_.shutdown(asio::ip::tcp::socket::shutdown_send);
454 r = rpc_result<R, coro_rpc_protocol>{
455 unexpect_t{}, coro_rpc_protocol::rpc_error{std::errc::io_error,
456 ret.first.message()}};
457 co_return r;
458 }
459 else {
460 ret = co_await asio_util::async_write(
461 socket, asio::buffer(buffer.data(), buffer.size()));
462 }
463#else
464 ret = co_await asio_util::async_write(
465 socket, asio::buffer(buffer.data(), buffer.size()));
466#endif
467 if (!ret.first) {
468#ifdef UNIT_TEST_INJECT
469 if (g_action == inject_action::client_close_socket_after_send_payload) {
470 ELOGV(INFO, "client_id %d client_close_socket_after_send_payload",
471 client_id_);
472 r = rpc_result<R, coro_rpc_protocol>{
473 unexpect_t{}, coro_rpc_protocol::rpc_error{std::errc::io_error,
474 ret.first.message()}};
475 close();
476 co_return r;
477 }
478#endif
479 coro_rpc_protocol::resp_header header;
480 ret = co_await asio_util::async_read(
481 socket,
482 asio::buffer((char *)&header, coro_rpc_protocol::RESP_HEAD_LEN));
483 if (!ret.first) {
484 uint32_t body_len = header.length;
485 if (body_len > read_buf_.size()) {
486 read_buf_.resize(body_len);
487 }
488 ret = co_await asio_util::async_read(
489 socket, asio::buffer(read_buf_.data(), body_len));
490 if (!ret.first) {
491#ifdef GENERATE_BENCHMARK_DATA
492 std::ofstream file(
493 benchmark_file_path + std::string{get_func_name<func>()} + ".out",
494 std::ofstream::binary | std::ofstream::out);
495 file << std::string_view{(char *)&header,
496 coro_rpc_protocol::RESP_HEAD_LEN};
497 file << std::string_view{(char *)read_buf_.data(), body_len};
498 file.close();
499#endif
500 r = handle_response_buffer<R>(read_buf_.data(), ret.second,
501 std::errc{header.err_code});
502 if (!r) {
503 close();
504 }
505 co_return r;
506 }
507 }
508 }
509#ifdef UNIT_TEST_INJECT
510 if (g_action == inject_action::force_inject_client_write_data_timeout) {
511 is_timeout_ = true;
512 }
513#endif
514 if (is_timeout_) {
515 r = rpc_result<R, coro_rpc_protocol>{
516 unexpect_t{}, coro_rpc_protocol::rpc_error{
517 .code = std::errc::timed_out, .msg = {}}};
518 }
519 else {
520 r = rpc_result<R, coro_rpc_protocol>{
521 unexpect_t{},
522 coro_rpc_protocol::rpc_error{.code = std::errc::io_error,
523 .msg = ret.first.message()}};
524 }
525 close();
526 co_return r;
527 }
528 /*
529 * buffer layout
530 * ┌────────────────┬────────────────┐
531 * │req_header │args │
532 * ├────────────────┼────────────────┤
533 * │REQ_HEADER_LEN │variable length │
534 * └────────────────┴────────────────┘
535 */
536 template <auto func, typename... Args>
537 std::vector<std::byte> prepare_buffer(Args &&...args) {
538 std::vector<std::byte> buffer;
539 std::size_t offset = coro_rpc_protocol::REQ_HEAD_LEN;
540 if constexpr (sizeof...(Args) > 0) {
541 using arg_types = function_parameters_t<decltype(func)>;
542 pack_to<arg_types>(buffer, offset, std::forward<Args>(args)...);
543 }
544 else {
545 buffer.resize(offset);
546 }
547
548 auto &header = *(coro_rpc_protocol::req_header *)buffer.data();
549 header.magic = coro_rpc_protocol::magic_number;
550 header.function_id = func_id<func>();
551#ifdef UNIT_TEST_INJECT
552 header.seq_num = client_id_;
553 if (g_action == inject_action::client_send_bad_magic_num) {
554 header.magic = coro_rpc_protocol::magic_number + 1;
555 }
556 if (g_action == inject_action::client_send_header_length_0) {
557 header.length = 0;
558 }
559 else {
560#endif
561 header.length = buffer.size() - coro_rpc_protocol::REQ_HEAD_LEN;
562#ifdef UNIT_TEST_INJECT
563 }
564#endif
565 return buffer;
566 }

568 template <typename T>
569 rpc_result<T, coro_rpc_protocol> handle_response_buffer(
570 const std::byte *buffer, std::size_t len, std::errc rpc_errc) {
571 rpc_return_type_t<T> ret;
572 struct_pack::errc ec;
573 coro_rpc_protocol::rpc_error err;
574 if (rpc_errc == std::errc{}) {
575 ec = struct_pack::deserialize_to(ret, (const char *)buffer, len);
576 if (ec == struct_pack::errc::ok) {
577 if constexpr (std::is_same_v<T, void>) {
578 return {};
579 }
580 else {
581 return std::move(ret);
582 }
583 }
584 }
585 else {
586 err.code = rpc_errc;
587 ec = struct_pack::deserialize_to(err.msg, (const char *)buffer, len);
588 if (ec == struct_pack::errc::ok) {
589 return rpc_result<T, coro_rpc_protocol>{unexpect_t{}, std::move(err)};
590 }
591 }
592 // deserialize failed.
593 err = {std::errc::invalid_argument,
594 "failed to deserialize rpc return value"};
595 return rpc_result<T, coro_rpc_protocol>{unexpect_t{}, std::move(err)};
596 }

643 void stop_inner_io_context() {
644 if (thd_.joinable()) {
645 work_ = nullptr;
646 if (thd_.get_id() == std::this_thread::get_id()) {
647 // we are now running in inner_io_context_, so destruction it in another
648 // thread
649 std::thread thrd{[ioc = std::move(inner_io_context_),
650 thd = std::move(thd_)]() mutable {
651 thd.join();
652 }};
653 thrd.detach();
654 }
655 else {
656 thd_.join();
657 }
658 }
659
660 return;
661 }
