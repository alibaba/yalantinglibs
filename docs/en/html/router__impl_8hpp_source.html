56inline async_simple::coro::Lazy<std::pair<std::errc, std::vector<char>>>
57router::route_coro(auto handler, std::string_view data, rpc_conn conn) {
58 uint32_t id = *(uint32_t *)data.data();
59 if (handler) [[likely]] {
60 try {
61#ifndef NDEBUG
62 if (auto it = id2name_.find(id); it != id2name_.end()) {
63 ELOGV(INFO, "route coro function name %d", it->second.data());
64 }
65#endif
66 co_return co_await (*handler)(data, conn);
67 } catch (const std::exception &e) {
68 ELOGV(ERROR, "the rpc function has exception %s, function id %d",
69 e.what(), id);
70 co_return pack_result(std::errc::interrupted, e.what());
71 } catch (...) {
72 ELOGV(ERROR, "the rpc function has unknown exception, function id %d",
73 id);
74 co_return pack_result(std::errc::interrupted, "unknown exception");
75 }
76 }
77 else [[unlikely]] {
78 ELOGV(ERROR, "the rpc function not found, function id %d", id);
79 co_return pack_result(std::errc::function_not_supported,
80 "the function not found");
81 }
82}

112template <auto func, typename Self>
113inline void router::regist_one_handler(Self *self) {
114 if (self == nullptr) [[unlikely]] {
115 ELOGV(CRITICAL, "null connection!");
116 }
117
118 constexpr auto name = get_func_name<func>();
119 constexpr auto id =
120 struct_pack::MD5::MD5Hash32Constexpr(name.data(), name.length());
121 using return_type = function_return_type_t<decltype(func)>;
122 if constexpr (is_specialization_v<return_type, async_simple::coro::Lazy>) {
123 auto it = coro_handlers_.emplace(
124 id,
125 [self](std::string_view data, rpc_conn conn)
126 -> async_simple::coro::Lazy<
127 std::pair<std::errc, std::vector<char>>> {
128 co_return co_await internal::execute_coro<func>(data, conn, self);
129 });
130 if (!it.second) {
131 ELOGV(CRITICAL, "duplication function %s register!", name.data());
132 }
133 }
134 else {
135 auto it = handlers_.emplace(
136 id, [self](std::string_view data, rpc_conn conn) mutable {
137 return internal::execute<func>(data, conn, self);
138 });
139 if (!it.second) {
140 ELOGV(CRITICAL, "duplication function %s register!", name.data());
141 }
142 }
143
144 id2name_.emplace(id, name);
145}

180template <auto func>
181inline bool router::remove_handler() {
182 constexpr auto name = get_func_name<func>();
183 constexpr auto id =
184 struct_pack::MD5::MD5Hash32Constexpr(name.data(), name.length());
185
186 id2name_.erase(id);
187
188 auto it = handlers_.find(id);
189 if (it != handlers_.end()) {
190 return handlers_.erase(id);
191 }
192 else {
193 auto coro_it = coro_handlers_.find(id);
194 if (coro_it != coro_handlers_.end()) {
195 return coro_handlers_.erase(id);
196 }
197
198 return false;
199 }
200}
