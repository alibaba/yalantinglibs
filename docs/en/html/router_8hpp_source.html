97 template <auto func, typename Self>
98 void regist_one_handler(Self *self) {
99 if (self == nullptr) [[unlikely]] {
100 ELOGV(CRITICAL, "null connection!");
101 }
102
103 constexpr auto name = get_func_name<func>();
104 constexpr auto id =
105 struct_pack::MD5::MD5Hash32Constexpr(name.data(), name.length());
106 using return_type = function_return_type_t<decltype(func)>;
107 if constexpr (is_specialization_v<return_type, async_simple::coro::Lazy>) {
108 auto it = coro_handlers_.emplace(
109 id,
110 [self](
111 std::string_view data, rpc_conn conn,
112 typename rpc_protocol::req_header &req_head,
113 typename rpc_protocol::supported_serialize_protocols protocols) {
114 execute_visitor<func, Self> visitor{data, std::move(conn), req_head,
115 self};
116 return std::visit(visitor, protocols);
117 });
118 if (!it.second) {
119 ELOGV(CRITICAL, "duplication function %s register!", name.data());
120 }
121 }
122 else {
123 auto it = handlers_.emplace(
124 id,
125 [self](
126 std::string_view data, rpc_conn conn,
127 typename rpc_protocol::req_header &req_head,
128 typename rpc_protocol::supported_serialize_protocols protocols) {
129 return std::visit(
130 [data, conn = std::move(conn), self,
131 &req_head]<typename serialize_protocol>(
132 const serialize_protocol &obj) mutable {
133 return internal::execute<rpc_protocol, serialize_protocol,
134 func>(data, std::move(conn),
135 req_head, self);
136 },
137 protocols);
138 });
139 if (!it.second) {
140 ELOGV(CRITICAL, "duplication function %s register!", name.data());
141 }
142 }
143
144 id2name_.emplace(id, name);
145 }

154 constexpr auto id =
155 struct_pack::MD5::MD5Hash32Constexpr(name.data(), name.length());
156 if constexpr (is_specialization_v<return_type, async_simple::coro::Lazy>) {
157 auto it = coro_handlers_.emplace(id, [](std::string_view data,
158 rpc_conn conn,
159 typename rpc_protocol::req_header
160 &req_head,
161 typename rpc_protocol::
162 supported_serialize_protocols
163 protocols) {
164 execute_visitor<func, void> visitor{data, std::move(conn), req_head};
165 return std::visit(visitor, protocols);
166 });
167 if (!it.second) {

170 }

175 typename rpc_protocol::req_header &req_head,
176 typename rpc_protocol::supported_serialize_protocols protocols) {

209 async_simple::coro::Lazy<std::pair<std::errc, std::string>> route_coro(
210 auto handler, std::string_view data, rpc_conn conn,
211 typename rpc_protocol::req_header &header,
212 typename rpc_protocol::supported_serialize_protocols protocols,
213 const typename rpc_protocol::route_key_t &route_key) {
214 using namespace std::string_literals;
215 if (handler) [[likely]] {
216 try {
217#ifndef NDEBUG
218 ELOGV(INFO, "route function name: %s", get_name(route_key).data());
219
220#endif
221 // clang-format off
222 auto res = co_await (*handler)(data, std::move(conn), header, protocols);
223 // clang-format on
224 if (res.has_value()) [[likely]] {
225 co_return std::make_pair(std::errc{}, std::move(res.value()));
226 }
227 else { // deserialize failed
228 ELOGV(ERROR, "payload deserialize failed in rpc function: %s",
229 get_name(route_key).data());
230 co_return std::make_pair(std::errc::invalid_argument,
231 "invalid rpc function arguments"s);
232 }
233 } catch (const std::exception &e) {
234 ELOGV(ERROR, "exception: %s in rpc function: %s", e.what(),
235 get_name(route_key).data());
236 co_return std::make_pair(std::errc::interrupted, e.what());
237 } catch (...) {
238 ELOGV(ERROR, "unknown exception in rpc function: %s",
239 get_name(route_key).data());
240 co_return std::make_pair(std::errc::interrupted, "unknown exception"s);
241 }
242 }
243 else {
244 std::ostringstream ss;
245 ss << route_key;
246 ELOGV(ERROR, "the rpc function not registered, function ID: %s",
247 ss.str().data());
248 co_return std::make_pair(std::errc::function_not_supported,
249 "the rpc function not registered"s);
250 }
251 }
