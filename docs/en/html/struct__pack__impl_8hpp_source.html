101template <typename T, uint64_t version>
102struct compatible : public std::optional<T> {
103 constexpr compatible() = default;
104 constexpr compatible(const compatible &other) = default;
105 constexpr compatible(compatible &&other) = default;

113 using base::base;

115 const compatible<T, version> &other) {
116 return static_cast<bool>(self) == static_cast<bool>(other) &&
117 (!self || *self == *other);
118 }
119 static constexpr uint64_t version_number = version;
120};

183template <class U>
184constexpr auto get_types(U &&t) {
185 using T = std::remove_cvref_t<U>;
186 if constexpr (std::is_fundamental_v<T> || std::is_enum_v<T> || varint_t<T> ||
187 std::is_same_v<std::string, T> || container<T> || optional<T> ||
188 unique_ptr<T> || variant<T> || expected<T> || array<T> ||
189 c_array<T> || std::is_same_v<std::monostate, T>) {
190 return std::tuple<T>{};
191 }
192 else if constexpr (tuple<T>) {
193 return T{};
194 }
195 else if constexpr (is_trivial_tuple<T>) {
196 return T{};
197 }
198 else if constexpr (pair<T>) {
199 return std::tuple<typename T::first_type, typename T::second_type>{};
200 }
201 else if constexpr (std::is_aggregate_v<T>) {
202 // clang-format off
203 return visit_members(
204 std::forward<U>(t), [&]<typename... Args>(Args &&
205 ...) CONSTEXPR_INLINE_LAMBDA {
206 return std::tuple<std::remove_cvref_t<Args>...>{};
207 });
208 // clang-format on
209 }
210 else {
211 static_assert(!sizeof(T), "the type is not supported!");
212 }
213}

215template <typename T>
216struct is_trivial_serializable {
217 private:
218 static constexpr bool solve() {
219 if constexpr (std::is_enum_v<T> || std::is_fundamental_v<T>) {
220 return true;
221 }
222 else if constexpr (array<T>) {
223 return is_trivial_serializable<typename T::value_type>::value;
224 }
225 else if constexpr (c_array<T>) {
226 return is_trivial_serializable<
227 typename std::remove_all_extents<T>::type>::value;
228 }
229 else if constexpr (!pair<T> && tuple<T> && !is_trivial_tuple<T>) {
230 return false;
231 }
232 else if constexpr (container<T> || optional<T> || variant<T> ||
233 unique_ptr<T> || expected<T> || container_adapter<T> ||
234 varint_t<T>) {
235 return false;
236 }
237 else if constexpr (pair<T>) {
238 return is_trivial_serializable<typename T::first_type>::value &&
239 is_trivial_serializable<typename T::second_type>::value;
240 }
241 else if constexpr (is_trivial_tuple<T>) {
242 return []<std::size_t... I>(std::index_sequence<I...>)
243 CONSTEXPR_INLINE_LAMBDA {
244 return (is_trivial_serializable<std::tuple_element_t<I, T>>::value &&
245 ...);
246 }
247 (std::make_index_sequence<std::tuple_size_v<T>>{});
248 }
249 else if constexpr (std::is_class_v<T>) {
250 using T_ = decltype(get_types(T{}));
251 return []<std::size_t... I>(std::index_sequence<I...>)
252 CONSTEXPR_INLINE_LAMBDA {
253 return (is_trivial_serializable<std::tuple_element_t<I, T_>>::value &&
254 ...);
255 }
256 (std::make_index_sequence<std::tuple_size_v<T_>>{});
257 }
258 else
259 return false;
260 }
261
262 public:
263 static inline constexpr bool value = is_trivial_serializable::solve();
264};

351template <typename T>
352consteval type_id get_integral_type() {
353 if constexpr (std::is_same_v<int32_t, T>) {
354 return type_id::int32_t;
355 }
356 else if constexpr (std::is_same_v<uint32_t, T>) {
357 return type_id::uint32_t;
358 }
359 else if constexpr (std::is_same_v<int64_t, T> ||
360 (sizeof(long long) == 8 && std::is_same_v<T, long long>)) {
361 return type_id::int64_t;
362 }
363 else if constexpr (std::is_same_v<uint64_t, T> ||
364 (sizeof(unsigned long long) == 8 &&
365 std::is_same_v<T, unsigned long long>)) {
366 return type_id::uint64_t;
367 }
368 else if constexpr (std::is_same_v<int8_t, T> ||
369 std::is_same_v<signed char, T>) {
370 return type_id::int8_t;
371 }
372 else if constexpr (std::is_same_v<uint8_t, T> ||
373 std::is_same_v<unsigned char, T>) {
374 return type_id::uint8_t;
375 }
376 else if constexpr (std::is_same_v<int16_t, T>) {
377 return type_id::int16_t;
378 }
379 else if constexpr (std::is_same_v<uint16_t, T>) {
380 return type_id::uint16_t;
381 }
382 // In struct_pack, the char will be saved as unsigned!
383 else if constexpr (std::is_same_v<char, T> || std::is_same_v<char8_t, T>) {
384 return type_id::char_8_t;
385 }
386#ifdef STRUCT_PACK_ENABLE_UNPORTABLE_TYPE
387 else if constexpr (std::is_same_v<wchar_t, T>) {
388 return type_id::w_char_t;
389 }
390#endif
391 // char16_t's size maybe bigger than 16 bits, which is not supported.
392 else if constexpr (std::is_same_v<char16_t, T>) {
393 static_assert(sizeof(char16_t) == 2,
394 "sizeof(char16_t)!=2, which is not supported.");
395 return type_id::char_16_t;
396 }
397 // char32_t's size maybe bigger than 32bits, which is not supported.
398 else if constexpr (std::is_same_v<char32_t, T> && sizeof(char32_t) == 4) {
399 static_assert(sizeof(char32_t) == 4,
400 "sizeof(char16_t)!=4, which is not supported.");
401 return type_id::char_32_t;
402 }
403 else if constexpr (std::is_same_v<bool, T> && sizeof(bool)) {
404 static_assert(sizeof(bool) == 1,
405 "sizeof(bool)!=1, which is not supported.");
406 return type_id::bool_t;
407 }
408 else {
409 /*
410 * Due to different data model,
411 * the following types are not allowed on macOS
412 * but work on Linux
413 * For example,
414 * on macOS, `typedef unsigned long long uint64_t;`
415 * on Linux, `typedef unsigned long int uint64_t;`
416 *
417 * - long
418 * - long int
419 * - signed long
420 * - signed long int
421 * - unsigned long
422 * - unsigned long int
423 *
424 * We add this static_assert to give more information about not supported
425 * type.
426 */
427 static_assert(
428 !std::is_same_v<wchar_t, T>,
429 "Tips: Add macro STRUCT_PACK_ENABLE_UNPORTABLE_TYPE to support "
430 "wchar_t");
431 static_assert(!std::is_same_v<long, T> && !std::is_same_v<unsigned long, T>,
432 "The long types have different width in "
433 "different data model. "
434 "see "
435 "https://en.cppreference.com/w/cpp/language/"
436 "types. "
437 "Please use fixed width integer types. e.g. "
438 "int32_t, int64_t. "
439 "see "
440 "https://en.cppreference.com/w/cpp/types/"
441 "integer.");
442 static_assert(!sizeof(T), "not supported type");
443 // This branch will always compiled error.
444 }

447template <typename T>
448consteval type_id get_floating_point_type() {
449 if constexpr (std::is_same_v<float, T>) {
450 if constexpr (!std::numeric_limits<float>::is_iec559 ||
451 sizeof(float) != 4) {
452 static_assert(
453 !sizeof(T),
454 "The float type in this machine is not standard IEEE 754 32bits "
455 "float point!");
456 }
457 return type_id::float32_t;
458 }
459 else if constexpr (std::is_same_v<double, T>) {
460 if constexpr (!std::numeric_limits<double>::is_iec559 ||
461 sizeof(double) != 8) {
462 static_assert(
463 !sizeof(T),
464 "The double type in this machine is not standard IEEE 754 64bits "
465 "float point!");
466 }
467 return type_id::float64_t;
468 }
469 else if constexpr (std::is_same_v<long double, T>) {
470 if constexpr (sizeof(long double) != 16 ||
471 std::numeric_limits<long double>::is_iec559) {
472 static_assert(!sizeof(T),
473 "The long double type in this machine is not standard IEEE "
474 "754 128bits "
475 "float point!");
476 }
477 return type_id::float128_t;
478 }
479 else {
480 static_assert(!sizeof(T), "not supported type");
481 }

484template <typename T>
485consteval type_id get_type_id() {
486 static_assert(CHAR_BIT == 8);
487 // compatible member, which should be ignored in MD5 calculated.
488 if constexpr (optional<T> && is_compatible<T>) {
489 return type_id::compatible_t;
490 }
491 else if constexpr (std::is_enum_v<T>) {
492 return get_integral_type<std::underlying_type_t<T>>();
493 }
494 else if constexpr (std::is_integral_v<T>) {
495 return get_integral_type<T>();
496 }
497 else if constexpr (std::is_floating_point_v<T>) {
498 return get_floating_point_type<T>();
499 }
500 else if constexpr (detail::varint_t<T>) {
501 return get_varint_type<T>();
502 }
503 else if constexpr (std::is_same_v<T, std::monostate> ||
504 std::is_same_v<T, void>) {
505 return type_id::monostate_t;
506 }
507 else if constexpr (string<T>) {
508 return type_id::string_t;
509 }
510 else if constexpr (array<T> || c_array<T>) {
511 return type_id::array_t;
512 }
513 else if constexpr (map_container<T>) {
514 return type_id::map_container_t;
515 }
516 else if constexpr (set_container<T>) {
517 return type_id::set_container_t;
518 }
519 else if constexpr (container<T>) {
520 return type_id::container_t;
521 }
522 else if constexpr (optional<T> || unique_ptr<T>) {
523 return type_id::optional_t;
524 }
525 else if constexpr (variant<T>) {
526 static_assert(
527 std::variant_size_v<T> > 1 ||
528 (std::variant_size_v<T> == 1 &&
529 !std::is_same_v<std::variant_alternative_t<0, T>, std::monostate>),
530 "The variant should contain's at least one type!");
531 static_assert(std::variant_size_v<T> < 256, "The variant is too complex!");
532 return type_id::variant_t;
533 }
534 else if constexpr (expected<T>) {
535 return type_id::expected_t;
536 }
537 else if constexpr (is_trivial_tuple<T> || pair<T>) {
538 return type_id::trivial_class_t;
539 }
540 else if constexpr (tuple<T>) {
541 return type_id::non_trivial_class_t;
542 }
543 else if constexpr (std::is_class_v<T>) {
544 static_assert(std::is_aggregate_v<std::remove_cvref_t<T>>);
545 return type_id::trivial_class_t;
546 }
547 else {
548 static_assert(!sizeof(T), "not supported type");
549 }

622template <typename arg, typename... ParentArgs>
623consteval std::size_t check_cycle() {
624 using types_tuple = std::tuple<ParentArgs...>;
625 if constexpr (sizeof...(ParentArgs)) {
626 return []<std::size_t... I>(std::index_sequence<I...>) {
627 std::size_t ret = (std::max)(
628 {(std::is_same_v<std::tuple_element_t<I, types_tuple>, arg> ? I + 1
629 : 0)...});
630 return ret;
631 }
632 (std::make_index_sequence<sizeof...(ParentArgs)>());
633 }
634 else {
635 return 0;
636 }

650template <typename Arg, typename... ParentArgs>
651consteval decltype(auto) get_type_literal() {
652 constexpr auto id = get_type_id<Arg>();
653 constexpr auto ret = string_literal<char, 1>{{static_cast<char>(id)}};
654 if constexpr (id == type_id::non_trivial_class_t ||
655 id == type_id::trivial_class_t) {
656 using Args = decltype(get_types(Arg{}));
657 constexpr auto body = get_type_literal<Args, Arg, ParentArgs...>(
658 std::make_index_sequence<std::tuple_size_v<Args>>());
659 if constexpr (is_trivial_serializable<Arg>::value) {
660 static_assert(
661 min_align<Arg>() == '0' || min_align<Arg>() <= max_align<Arg>(),
662 "#pragma pack may decrease the alignment of a class, however, "
663 "it cannot make a class over aligned.");
664 constexpr auto end =
665 string_literal<char, 3>{{static_cast<char>(min_align<Arg>()),
666 static_cast<char>(max_align<Arg>()),
667 static_cast<char>(type_id::type_end_flag)}};
668 return ret + body + end;
669 }
670 else {
671 constexpr auto end =
672 string_literal<char, 1>{{static_cast<char>(type_id::type_end_flag)}};
673 return ret + body + end;
674 }
675 }
676 else if constexpr (id == type_id::variant_t) {
677 constexpr auto sz = std::variant_size_v<Arg>;
678 static_assert(sz > 0, "empty param of std::variant is not allowed!");
679 static_assert(sz < 256, "too many alternative type in variant!");
680 constexpr auto body = get_variant_literal<Arg>(
681 std::make_index_sequence<std::variant_size_v<Arg>>());
682 constexpr auto end =
683 string_literal<char, 1>{{static_cast<char>(type_id::type_end_flag)}};
684 return ret + body + end;
685 }
686 else if constexpr (id == type_id::array_t) {
687 constexpr auto sz =
688 sizeof(Arg) /
689 sizeof(decltype(std::declval<
690 Arg>()[0])); // std::size(std::declval<Arg>());
691 return ret +
692 get_type_literal<
693 std::remove_cvref_t<decltype(std::declval<Arg>()[0])>, Arg,
694 ParentArgs...>() +
695 get_size_literal<sz>();
696 }
697 else if constexpr (unique_ptr<Arg>) {
698 // detect cycle in graph
699 constexpr std::size_t has_cycle =
700 check_cycle<typename Arg::element_type, ParentArgs...>();
701 if constexpr (has_cycle) {
702 return ret +
703 string_literal<char, 1>{

917template <typename T>
918constexpr size_info STRUCT_PACK_INLINE calculate_one_size(const T &item) {
919 constexpr auto id = get_type_id<std::remove_cvref_t<T>>();
920 static_assert(id != detail::type_id::type_end_flag);
921 using type = std::remove_cvref_t<decltype(item)>;
922 static_assert(!std::is_pointer_v<type>);
923 size_info ret{.total = 0, .size_cnt = 0, .max_size = 0};
924 if constexpr (std::is_same_v<type, std::monostate>) {
925 }
926 else if constexpr (std::is_fundamental_v<type> || std::is_enum_v<type>) {
927 ret.total = sizeof(type);
928 }
929 else if constexpr (detail::varint_t<type>) {
930 ret.total = detail::calculate_varint_size(item);
931 }
932 else if constexpr (c_array<type> || array<type>) {
933 if constexpr (is_trivial_serializable<type>::value) {
934 ret.total = sizeof(type);
935 }
936 else {
937 for (auto &i : item) {
938 ret += calculate_one_size(i);
939 }
940 }
941 }
942 else if constexpr (container<type>) {
943 ret.size_cnt += 1;
944 ret.max_size = (std::max)(ret.max_size, item.size());
945 if constexpr (trivially_copyable_container<type>) {
946 using value_type = typename type::value_type;
947 ret.total = item.size() * sizeof(value_type);
948 }
949 else {
950 for (auto &&i : item) {
951 ret += calculate_one_size(i);
952 }
953 }
954 }
955 else if constexpr (container_adapter<type>) {
956 static_assert(!sizeof(type), "the container adapter type is not supported");
957 }
958 else if constexpr (!pair<type> && tuple<type> && !is_trivial_tuple<type>) {
959 std::apply(
960 [&](auto &&...items) CONSTEXPR_INLINE_LAMBDA {
961 ret += calculate_payload_size(items...);
962 },
963 item);
964 }
965 else if constexpr (optional<type>) {
966 ret.total = sizeof(char);
967 if (item.has_value()) {
968 ret += calculate_one_size(*item);
969 }
970 }
971 else if constexpr (variant<type>) {
972 ret.total = sizeof(uint8_t);
973 ret += std::visit(
974 [](const auto &e) {
975 return calculate_one_size(e);
976 },
977 item);
978 }
979 else if constexpr (expected<type>) {
980 ret.total = sizeof(bool);
981 if (item.has_value()) {
982 if constexpr (!std::is_same_v<typename type::value_type, void>)
983 ret += calculate_one_size(item.value());
984 }
985 else {
986 ret += calculate_one_size(item.error());
987 }
988 }
989 else if constexpr (std::is_class_v<type>) {
990 if constexpr (!pair<type> && !is_trivial_tuple<type>)
991 static_assert(std::is_aggregate_v<std::remove_cvref_t<type>>);
992 if constexpr (is_trivial_serializable<type>::value) {
993 ret.total = sizeof(type);
994 }
995 else {
996 visit_members(item, [&](auto &&...items) CONSTEXPR_INLINE_LAMBDA {
997 ret += calculate_payload_size(items...);
998 });
999 }
1000 }
1001 else {
1002 static_assert(!sizeof(type), "the type is not supported yet");
1003 }
1004 return ret;
1005}

1305 constexpr bool has_compile_time_determined_meta_info =
1306 has_compatible || has_type_literal;
1307 if constexpr (has_compile_time_determined_meta_info) {
1308 ret.len_ += sizeof(unsigned char);
1309 }
1310 }
1311 else {
1312 if (sz_info.max_size < (int64_t{1} << 16)) {
1313 ret.len_ += sz_info.total + sz_info.size_cnt * 2;
1314 ret.metainfo_ = 0b01000;
1315 }
1316 else if (sz_info.max_size < (int64_t{1} << 32)) {
1317 ret.len_ += sz_info.total + sz_info.size_cnt * 4;
1318 ret.metainfo_ = 0b10000;
1319 }
1320 else {
1321 ret.len_ += sz_info.total + sz_info.size_cnt * 8;
1322 ret.metainfo_ = 0b11000;
1323 }
1324 // size_type >= 1 , has metainfo
1325 ret.len_ += sizeof(unsigned char);
1326 }
1327 if constexpr (has_type_literal) {
1328 constexpr auto type_literal = struct_pack::get_type_literal<Args...>();
1329 // struct_pack::get_type_literal<Args...>().size() crash in clang13. Bug?
1330 ret.len_ += type_literal.size() + 1;
1331 ret.metainfo_ |= 0b100;
1332 }
1333 if constexpr (has_compatible) { // calculate bytes count of serialize
1334 // length
1335 if (ret.len_ + 2 < (int64_t{1} << 16)) [[likely]] {
1336 ret.len_ += 2;
1337 ret.metainfo_ |= 0b01;
1338 }
1339 else if (ret.len_ + 4 < (int64_t{1} << 32)) {
1340 ret.len_ += 4;
1341 ret.metainfo_ |= 0b10;
1342 }
1343 else {
1344 ret.len_ += 8;
1345 ret.metainfo_ |= 0b11;
1346 }
1347 }
1348 return ret;
1349}

1434 template <std::size_t size_type, uint64_t version>
1435 constexpr void STRUCT_PACK_INLINE serialize_one(const auto &item) {
1436 using type = std::remove_cvref_t<decltype(item)>;
1437 static_assert(!std::is_pointer_v<type>);
1438 constexpr auto id = get_type_id<type>();
1439 if constexpr (version == UINT64_MAX) {
1440 if constexpr (id == type_id::compatible_t) {
1441 // do nothing
1442 }
1443 else if constexpr (std::is_same_v<type, std::monostate>) {
1444 // do nothing
1445 }
1446 else if constexpr (std::is_fundamental_v<type> || std::is_enum_v<type>) {
1447 writer_.write((char *)&item, sizeof(type));
1448 }
1449 else if constexpr (detail::varint_t<type>) {
1450 detail::serialize_varint(writer_, item);
1451 }
1452 else if constexpr (c_array<type> || array<type>) {
1453 if constexpr (is_trivial_serializable<type>::value) {
1454 writer_.write((char *)&item, sizeof(type));
1455 }
1456 else {
1457 for (const auto &i : item) {
1458 serialize_one<size_type, version>(i);
1459 }
1460 }
1461 }
1462 else if constexpr (map_container<type> || container<type>) {
1463 uint64_t size = item.size();
1464#ifdef STRUCT_PACK_OPTIMIZE
1465 writer_.write((char *)&size, size_type);
1466#else
1467 if constexpr (size_type == 1) {
1468 writer_.write((char *)&size, size_type);
1469 }
1470 else {
1471 switch ((info.metainfo() & 0b11000) >> 3) {
1472 case 1:
1473 writer_.write((char *)&size, 2);
1474 break;
1475 case 2:
1476 writer_.write((char *)&size, 4);
1477 break;
1478 case 3:
1479 writer_.write((char *)&size, 8);
1480 break;
1481 default:
1482 unreachable();
1483 }
1484 }
1485#endif
1486 if constexpr (trivially_copyable_container<type>) {
1487 using value_type = typename type::value_type;
1488 auto container_size = 1ull * size * sizeof(value_type);
1489 if (container_size >= PTRDIFF_MAX) [[unlikely]]
1490 unreachable();
1491 else {
1492 writer_.write((char *)item.data(), container_size);
1493 }
1494 return;
1495 }
1496 else {
1497 for (const auto &i : item) {
1498 serialize_one<size_type, version>(i);
1499 }
1500 }
1501 }
1502 else if constexpr (container_adapter<type>) {
1503 static_assert(!sizeof(type),
1504 "the container adapter type is not supported");
1505 }
1506 else if constexpr (!pair<type> && tuple<type> &&
1507 !is_trivial_tuple<type>) {
1508 std::apply(
1509 [&](auto &&...items) CONSTEXPR_INLINE_LAMBDA {
1510 serialize_many<size_type, version>(items...);
1511 },
1512 item);
1513 }
1514 else if constexpr (optional<type>) {
1515 bool has_value = item.has_value();
1516 writer_.write((char *)&has_value, sizeof(bool));
1517 if (has_value) {
1518 serialize_one<size_type, version>(*item);
1519 }
1520 }
1521 else if constexpr (variant<type>) {
1522 static_assert(std::variant_size_v<type> < 256,
1523 "variant's size is too large");
1524 uint8_t index = item.index();
1525 writer_.write((char *)&index, sizeof(index));
1526 std::visit(
1527 [this](auto &&e) {
1528 this->serialize_one<size_type, version>(e);
1529 },
1530 item);
1531 }
1532 else if constexpr (expected<type>) {
1533 bool has_value = item.has_value();
1534 writer_.write((char *)&has_value, sizeof(has_value));
1535 if (has_value) {
1536 if constexpr (!std::is_same_v<typename type::value_type, void>)
1537 serialize_one<size_type, version>(item.value());
1538 }
1539 else {
1540 serialize_one<size_type, version>(item.error());
1541 }
1542 }
1543 else if constexpr (std::is_class_v<type>) {
1544 if constexpr (!pair<type> && !is_trivial_tuple<type>)
1545 static_assert(std::is_aggregate_v<std::remove_cvref_t<type>>);
1546 if constexpr (is_trivial_serializable<type>::value) {
1547 writer_.write((char *)&item, sizeof(type));
1548 }
1549 else {
1550 visit_members(item, [&](auto &&...items) CONSTEXPR_INLINE_LAMBDA {
1551 serialize_many<size_type, version>(items...);
1552 });
1553 }
1554 }
1555 else {
1556 static_assert(!sizeof(type), "the type is not supported yet");
1557 }
1558 }
1559 else if constexpr (exist_compatible_member<type, version>) {
1560 if constexpr (id == type_id::compatible_t) {
1561 if constexpr (version == type::version_number) {
1562 bool has_value = item.has_value();
1563 writer_.write((char *)&has_value, sizeof(bool));
1564 if (has_value) {
1565 serialize_one<size_type, UINT64_MAX>(*item);
1566 }
1567 }
1568 }
1569 else if constexpr (c_array<type> || array<type>) {
1570 for (const auto &i : item) {
1571 serialize_one<size_type, version>(i);
1572 }
1573 }
1574 else if constexpr (map_container<type> || container<type>) {
1575 for (const auto &i : item) {
1576 serialize_one<size_type, version>(i);
1577 }
1578 }
1579 else if constexpr (!pair<type> && tuple<type> &&
1580 !is_trivial_tuple<type>) {
1581 std::apply(
1582 [&](auto &&...items) CONSTEXPR_INLINE_LAMBDA {
1583 serialize_many<size_type, version>(items...);
1584 },
1585 item);
1586 }
1587 else if constexpr (optional<type>) {
1588 if (item.has_value()) {
1589 serialize_one<size_type, version>(*item);
1590 }
1591 }
1592 else if constexpr (variant<type>) {
1593 std::visit(
1594 [this](const auto &e) {
1595 this->serialize_one<size_type, version>(e);
1596 },
1597 item);
1598 }
1599 else if constexpr (expected<type>) {
1600 if (item.has_value()) {
1601 if constexpr (!std::is_same_v<typename type::value_type, void>)
1602 serialize_one<size_type, version>(item.value());
1603 }
1604 else {
1605 serialize_one<size_type, version>(item.error());
1606 }
1607 }
1608 else if constexpr (std::is_class_v<type>) {
1609 visit_members(item, [&](auto &&...items) CONSTEXPR_INLINE_LAMBDA {
1610 serialize_many<size_type, version>(items...);
1611 });
1612 }
1613 }
1614 return;
1615 }

1680 return false;

1684 return true;
