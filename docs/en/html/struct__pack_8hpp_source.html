140template <typename... Args>
141STRUCT_PACK_INLINE consteval std::uint32_t get_type_code() {
142 static_assert(sizeof...(Args) > 0);
143 std::uint32_t ret;
144 if constexpr (sizeof...(Args) == 1) {
145 ret = detail::get_types_code<Args...,
146 decltype(detail::get_types<Args...>())>();
147 }
148 else {
149 ret = detail::get_types_code<std::tuple<std::remove_cvref_t<Args>...>,
150 std::tuple<Args...>>();
151 }
152 ret = ret - ret % 2;
153 return ret;
154}

192template <serialize_config conf = serialize_config{},
193 detail::struct_pack_buffer Buffer, typename... Args>
194void STRUCT_PACK_INLINE serialize_to(Buffer &buffer, const Args &...args) {
195 static_assert(sizeof...(args) > 0);
196 auto data_offset = buffer.size();
197 auto info = detail::get_serialize_runtime_info<conf>(args...);
198 auto total = data_offset + info.size();
199 buffer.resize(total);
200 auto writer =
201 struct_pack::detail::memory_writer{(char *)buffer.data() + data_offset};
202 struct_pack::detail::serialize_to<conf>(writer, info, args...);
203}

255template <typename T, typename... Args, struct_pack::reader_t Reader>
256[[nodiscard]] STRUCT_PACK_INLINE struct_pack::errc deserialize_to(
257 T &t, Reader &reader, Args &...args) {
258 detail::unpacker in(reader);
259 std::size_t consume_len;
260 auto old_pos = reader.tellg();
261 auto ret = in.deserialize_with_len(consume_len, t, args...);
262 std::size_t delta = reader.tellg() - old_pos;
263 if (ret == errc{}) [[likely]] {
264 if (consume_len > 0) [[likely]] {
265 if (delta > consume_len) [[unlikely]] {
266 ret = struct_pack::errc::invalid_buffer;
267 if constexpr (struct_pack::seek_reader_t<Reader>)
268 if (!reader.seekg(old_pos)) [[unlikely]] {
269 return struct_pack::errc::seek_failed;
270 }
271 }
272 else {
273 reader.ignore(consume_len - delta);
274 }
275 }
276 }
277 else {
278 if constexpr (struct_pack::seek_reader_t<Reader>)
279 if (!reader.seekg(old_pos)) [[unlikely]] {
280 return struct_pack::errc::seek_failed;
281 }
282 }
283 return ret;
284}
