102 template <typename... Args>
103 void response_msg(Args &&...args) {
104 if constexpr (std::is_same_v<return_msg_type, void>) {
105 static_assert(sizeof...(args) == 0, "illegal args");
106
107 auto old_flag = has_response_->exchange(true);
108 if (old_flag != false) {
109 ELOGV(ERROR, "response message more than one time");
110 return;
111 }
112
113 if (has_closed()) [[unlikely]] {
114 ELOGV(DEBUG, "response_msg failed: connection has been closed");
115 return;
116 }
117
118 response_handler_(std::move(conn_));
119 }
120 else {
121 static_assert(
122 requires { return_msg_type{std::forward<Args>(args)...}; },
123 "constructed return_msg_type failed by illegal args");
124 return_msg_type ret{std::forward<Args>(args)...};
125
126 auto old_flag = has_response_->exchange(true);
127 if (old_flag != false) {
128 ELOGV(ERROR, "response message more than one time");
129 return;
130 }
131
132 if (has_closed()) [[unlikely]] {
133 ELOGV(DEBUG, "response_msg failed: connection has been closed");
134 return;
135 }
136
137 response_handler_(std::move(conn_), std::move(ret));
138 }
139 }

166template <auto func>
167inline auto get_return_type() {
168 using T = decltype(func);
169 using param_type = function_parameters_t<T>;
170 using return_type = typename get_type_t<function_return_type_t<T>>::type;
171 if constexpr (std::is_void_v<param_type>) {
172 if constexpr (std::is_void_v<return_type>) {
173 return;
174 }
175 else {
176 return return_type{};
177 }
178 }
179 else {
180 using First = std::tuple_element_t<0, param_type>;
181 constexpr bool is_conn = is_specialization<First, context>::value;
182
183 if constexpr (is_conn) {
184 using U = typename First::return_type;
185 if constexpr (std::is_void_v<U>) {
186 return;
187 }
188 else {
189 return U{};
190 }
191 }
192 else if constexpr (!std::is_same_v<void, return_type>) {
193 return return_type{};
194 }
195 else {
196 return;
197 }
198 }
