115 template <typename Socket>
116 async_simple::coro::Lazy<void> start_impl(Socket &socket) noexcept {
117 char head_[RPC_HEAD_LEN];
118 auto self = shared_from_this();
119 while (true) {
120 reset_timer();
121 auto ret = co_await asio_util::async_read(
122 socket, asio::buffer(head_, RPC_HEAD_LEN));
123 cancel_timer();
124 // `co_await async_read` uses asio::async_read underlying.
125 // If eof occurred, the bytes_transferred of `co_await async_read` must
126 // less than RPC_HEAD_LEN. Incomplete data will be discarded.
127 // So, no special handling of eof is required.
128 if (ret.first) {
129 ELOGV(ERROR, "%s, %s", ret.first.message().data(), "read head error");
130 co_await close();
131 co_return;
132 }
133 assert(ret.second == RPC_HEAD_LEN);
134 auto errc = struct_pack::deserialize_to(header_, head_, RPC_HEAD_LEN);
135 if (errc != struct_pack::errc::ok) [[unlikely]] {
136 ELOGV(ERROR, "%s, %s",
137 std::make_error_code(std::errc::protocol_error).message().data(),
138 "deserialize error");
139 co_await close();
140 co_return;
141 }
142
143#ifdef UNIT_TEST_INJECT
144 client_id_ = header_.seq_num;
145 ELOGV(INFO, "conn_id %d, client_id %d", conn_id_, client_id_);
146#endif
147
148#ifdef UNIT_TEST_INJECT
149 if (g_action == inject_action::close_socket_after_read_header) {
150 ELOGV(WARN,
151 "inject action: close_socket_after_read_header, conn_id %d, "
152 "client_id %d",
153 conn_id_, client_id_);
154 co_await close();
155 co_return;
156 }
157#endif
158 if (header_.magic != magic_number) [[unlikely]] {
159 ELOGV(ERROR, "bad magic number, conn_id %d", conn_id_);
160 co_await close();
161 co_return;
162 }
163 if (header_.length == 0) [[unlikely]] {
164 ELOGV(ERROR, "bad length: %d, conn_id %d", header_.length, conn_id_);
165 co_await close();
166 co_return;
167 }
168
169 if (header_.length > body_size_) {
170 body_size_ = header_.length;
171 body_.resize(body_size_);
172 }
173
174 ret = co_await asio_util::async_read(
175 socket, asio::buffer(body_.data(), header_.length));
176 if (ret.first) [[unlikely]] {
177 ELOGV(ERROR, "read error: %s, conn_id %d", ret.first.message().data(),
178 conn_id_);
179 co_await close();
180 co_return;
181 }
182
183 std::pair<std::errc, std::vector<char>> pair{};
184 auto handler = router_.get_handler({body_.data(), ret.second});
185 if (!handler) {
186 auto coro_handler =
187 router_.get_coro_handler({body_.data(), ret.second});
188 pair = co_await router_.route_coro(coro_handler,
189 {body_.data(), ret.second}, self);
190 }
191 else {
192 pair = router_.route(handler, {body_.data(), ret.second}, self);
193 }
194
195 auto &[err, buf] = pair;
196 if (delay_) {
197 delay_ = false;
198 continue;
199 }
200
201 rpc_header resp_header = header_;
202 resp_header.length = buf.size() - RESPONSE_HEADER_LEN;
203 struct_pack::serialize_to(buf.data(), RESPONSE_HEADER_LEN, resp_header);
204
205#ifdef UNIT_TEST_INJECT
206 if (g_action == inject_action::close_socket_after_send_length) {
207 ELOGV(WARN,
208 "inject action: close_socket_after_send_length conn_id %d, "
209 "client_id %d",
210 conn_id_, client_id_);
211 co_await asio_util::async_write(
212 socket, asio::buffer(buf.data(), RESPONSE_HEADER_LEN));
213 co_await close();
214 co_return;
215 }
216 if (g_action == inject_action::server_send_bad_rpc_result) {
217 ELOGV(WARN,
218 "inject action: server_send_bad_rpc_result conn_id %d, client_id "
219 "%d",
220 conn_id_, client_id_);
221 buf[RESPONSE_HEADER_LEN + 1] = (buf[RESPONSE_HEADER_LEN + 1] + 1);
222 }
223#endif
224 if (rsp_err_ == err_ok) [[likely]] {
225 if (err != err_ok) [[unlikely]] {
226 rsp_err_ = err;
227 }
228 write_queue_.push_back(std::move(buf));
229 if (write_queue_.size() == 1) {
230 send_data().start([ec = shared_from_this()](auto &&) {
231 });
232 }
233 if (err != err_ok) [[unlikely]] {
234 co_return;
235 }
236 }
237 }

245 template <typename R>
246 void response_msg(const R &ret) {
247 if (has_closed()) [[unlikely]] {
248 ELOGV(DEBUG, "response_msg failed: connection has been closed");
249 return;
250 }
251
252 auto buf = struct_pack::serialize_with_offset(RESPONSE_HEADER_LEN, ret);
253 rpc_header resp_header = header_;
254 resp_header.length = buf.size() - RESPONSE_HEADER_LEN;
255 struct_pack::serialize_to(buf.data(), RPC_HEAD_LEN, resp_header);
256
257 auto self = shared_from_this();
258 response(std::move(buf), std::move(self)).via(&executor_).detach();
259 }

316 async_simple::coro::Lazy<void> response(std::vector<char> buf,
317 auto self) noexcept {
318 if (has_closed()) [[unlikely]] {
319 ELOGV(DEBUG, "response_msg failed: connection has been closed");
320 co_return;
321 }
322#ifdef UNIT_TEST_INJECT
323 if (g_action == inject_action::close_socket_after_send_length) {
324 ELOGV(WARN, "inject action: close_socket_after_send_length");
325 buf.resize(RESPONSE_HEADER_LEN);
326 }
327#endif
328 write_queue_.push_back(std::move(buf));
329 if (write_queue_.size() > 1) {
330 co_return;
331 }
332 co_await send_data();

335 async_simple::coro::Lazy<void> send_data(bool close_ssl = true) {
336 std::pair<std::error_code, size_t> ret;
337 while (!write_queue_.empty()) {
338 auto &msg = write_queue_.front();
339#ifdef UNIT_TEST_INJECT
340 if (g_action == inject_action::force_inject_connection_close_socket) {
341 ELOGV(
342 WARN,
343 "inject action: force_inject_connection_close_socket, conn_id %d, "
344 "client_id %d",
345 conn_id_, client_id_);
346 co_await close(false);
347 co_return;
348 }
349#endif
350#ifdef ENABLE_SSL
351 if (use_ssl_) {
352 assert(ssl_stream_);
353 ret = co_await asio_util::async_write(*ssl_stream_, asio::buffer(msg));
354 }
355 else {
356#endif
357 ret = co_await asio_util::async_write(socket_, asio::buffer(msg));
358#ifdef ENABLE_SSL
359 }
360#endif
361 if (ret.first) [[unlikely]] {
362 ELOGV(ERROR, "%s, %s", ret.first.message().data(), "async_write error");
363 co_await close(close_ssl);
364 co_return;
365 }
366 write_queue_.pop_front();
367 }
368 if (rsp_err_ != err_ok) [[unlikely]] {
369 ELOGV(ERROR, "%s, %s", std::make_error_code(rsp_err_).message().data(),
370 "rsp_err_");
371 co_await close(false);
372 co_return;
373 }
374#ifdef UNIT_TEST_INJECT
375 if (g_action == inject_action::close_socket_after_send_length) {
376 ELOGV(INFO,
377 "inject action: close_socket_after_send_length, conn_id %d, "
378 "client_id %d",
379 conn_id_, client_id_);
380 // Attention: close ssl stream after read error
381 // otherwise, server will crash
382 co_await close(false);
383 co_return;
384 }
385#endif
386 }

412 void reset_timer() {
413 if (!enable_check_timeout_) {
414 return;
415 }
416
417 timer_.expires_from_now(keep_alive_timeout_duration_);
418 timer_.async_wait(
419 [this, self = shared_from_this()](asio::error_code const &ec) {
420 if (ec) {
421 return;
422 }
423
424#ifdef UNIT_TEST_INJECT
425 ELOGV(INFO, "close timeout client_id %d conn_id %d", client_id_,
426 conn_id_);

428 ELOGV(INFO, "close timeout client conn_id %d", conn_id_);
429#endif

431 close_socket(false);
