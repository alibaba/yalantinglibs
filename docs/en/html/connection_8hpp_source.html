112 template <typename... Args>
113 void response_msg(Args &&...args) {
114 if constexpr (std::is_same_v<return_msg_type, void>) {
115 static_assert(sizeof...(args) == 0, "illegal args");
116
117 auto old_flag = has_response_->exchange(true);
118 if (old_flag != false) {
119 ELOGV(ERROR, "response message more than one time");
120 return;
121 }
122
123 if (has_closed()) [[unlikely]] {
124 ELOGV(DEBUG, "response_msg failed: connection has been closed");
125 return;
126 }
127
128 response_handler_(std::move(conn_));
129 }
130 else {
131 static_assert(
132 requires { return_msg_type{std::forward<Args>(args)...}; },
133 "constructed return_msg_type failed by illegal args");
134 return_msg_type ret{std::forward<Args>(args)...};
135
136 auto old_flag = has_response_->exchange(true);
137 if (old_flag != false) {
138 ELOGV(ERROR, "response message more than one time");
139 return;
140 }
141
142 if (has_closed()) [[unlikely]] {
143 ELOGV(DEBUG, "response_msg failed: connection has been closed");
144 return;
145 }
146
147 response_handler_(std::move(conn_), std::move(ret));
148 }
149 }

176template <auto func>
177inline auto get_return_type() {
178 using T = decltype(func);
179 using param_type = function_parameters_t<T>;
180 using return_type = typename get_type_t<function_return_type_t<T>>::type;
181 if constexpr (std::is_void_v<param_type>) {
182 if constexpr (std::is_void_v<return_type>) {
183 return;
184 }
185 else {
186 return return_type{};
187 }
188 }
189 else {
190 using First = std::tuple_element_t<0, param_type>;
191 constexpr bool is_conn = is_specialization<First, connection>::value;
192
193 if constexpr (is_conn) {
194 using U = typename First::return_type;
195 if constexpr (std::is_void_v<U>) {
196 return;
197 }
198 else {
199 return U{};
200 }
201 }
202 else if constexpr (!std::is_same_v<void, return_type>) {
203 return return_type{};
204 }
205 else {
206 return;
207 }
208 }
