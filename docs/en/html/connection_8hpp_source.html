84 void response_msg(Args &&...args) {
85 auto old_flag = has_response_->exchange(true);
86 if (old_flag != false) {
87 ELOGV(ERROR, "response message more than one time");

144inline auto get_return_type() {
145 using T = decltype(func);
146 using param_type = function_parameters_t<T>;
147 using return_type = typename get_type_t<function_return_type_t<T>>::type;
148 if constexpr (std::is_void_v<param_type>) {
149 if constexpr (std::is_void_v<return_type>) {
150 return;
151 }
152 else {
153 return return_type{};
154 }
155 }
156 else {
157 using First = std::tuple_element_t<0, param_type>;
158 constexpr bool is_conn = is_specialization<First, connection>::value;
159
160 if constexpr (is_conn) {
161 using U = typename First::return_type;
162 if constexpr (std::is_void_v<U>) {
163 return;

167 }
168 }
