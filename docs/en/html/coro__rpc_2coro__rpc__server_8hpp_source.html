105 [[nodiscard]] std::errc start() noexcept {
106 std::errc ec{};
107 {
108 std::unique_lock lock(start_mtx_);
109 if (flag_ != stat::init) {
110 if (flag_ == stat::started) {
111 ELOGV(INFO, "start again");
112 }
113 else if (flag_ == stat::stop) {
114 ELOGV(INFO, "has stoped");
115 }
116 return std::errc::io_error;
117 }
118
119 ec = listen();
120 if (ec == std::errc{}) {
121 thd_ = std::thread([this] {
122 pool_.run();
123 });
124 acceptor_thd_ = std::thread([this] {
125 acceptor_ioc_.run();
126 });
127
128 flag_ = stat::started;
129 }
130 else {
131 flag_ = stat::stop;
132 }
133 }
134
135 cond_.notify_all();
136
137 if (ec == std::errc{}) {
138 async_simple::coro::syncAwait(accept());
139 }
140 return ec;
141 }

191 void stop() {
192 std::unique_lock lock(start_mtx_);
193 if (flag_ == stat::stop) {
194 return;
195 }
196
197 ELOGV(INFO, "begin to stop coro_rpc_server, conn size %d", conns_.size());
198
199 close_acceptor();
200 if (flag_ == stat::started) {
201 {
202 std::unique_lock lock(conns_mtx_);
203 for (auto &conn : conns_) {
204 if (!conn.second->has_closed()) {
205 conn.second->async_close();
206 }
207 }
208
209 conns_.clear();
210 }
211
212 ELOGV(INFO, "wait for server's thread-pool finish all work.");
213 pool_.stop();
214 ELOGV(INFO, "server's thread-pool finished.");
215 }
216 if (thd_.joinable()) {
217 thd_.join();
218 }
219
220 if (acceptor_thd_.joinable()) {
221 ELOGV(INFO, "wait for server's acceptor thread finish all work.");
222 acceptor_thd_.join();
223 ELOGV(INFO, "server's acceptor thread finished.");
224 }
225 ELOGV(INFO, "stop coro_rpc_server ok");
226 flag_ = stat::stop;
227 }

306 std::errc listen() {
307 ELOGV(INFO, "begin to listen");
308 using asio::ip::tcp;
309 auto endpoint = tcp::endpoint(tcp::v4(), port_);
310 acceptor_.open(endpoint.protocol());
311#ifdef __GNUC__
312 acceptor_.set_option(tcp::acceptor::reuse_address(true));
313#endif
314 asio::error_code ec;
315 acceptor_.bind(endpoint, ec);
316 if (ec) {
317 ELOGV(ERROR, "bind port %d error : %s", port_.load(),
318 ec.message().data());
319 acceptor_.cancel(ec);
320 acceptor_.close(ec);
321 return std::errc::address_in_use;
322 }
323#ifdef _MSC_VER
324 acceptor_.set_option(tcp::acceptor::reuse_address(true));
325#endif
326 acceptor_.listen();
327
328 auto end_point = acceptor_.local_endpoint(ec);
329 if (ec) {
330 ELOGV(ERROR, "get local endpoint port %d error : %s", port_.load(),
331 ec.message().data());
332 return std::errc::address_in_use;
333 }
334 port_ = end_point.port();
335
336 ELOGV(INFO, "listen port %d successfully", port_.load());
337 return {};
338 }

340 async_simple::coro::Lazy<std::errc> accept() {
341 for (;;) {
342 auto &io_context = pool_.get_io_context();
343 asio::ip::tcp::socket socket(io_context);
344 auto error = co_await asio_util::async_accept(acceptor_, socket);
345#ifdef UNIT_TEST_INJECT
346 if (g_action == inject_action::force_inject_server_accept_error) {
347 asio::error_code ignored_ec;
348 socket.shutdown(asio::ip::tcp::socket::shutdown_both, ignored_ec);
349 socket.close(ignored_ec);
350 error = make_error_code(std::errc::io_error);
351 // only inject once
352 g_action = inject_action::nothing;
353 }
354#endif
355 if (error) {
356 ELOGV(ERROR, "accept failed, error: %s", error.message().data());
357 if (error == asio::error::operation_aborted) {
358 co_return std::errc::io_error;
359 }
360 continue;
361 }
362
363 int64_t conn_id = ++conn_id_;
364 ELOGV(INFO, "new client conn_id %d coming", conn_id);
365 auto conn = std::make_shared<coro_connection>(
366 io_context, std::move(socket), conn_timeout_duration_);
367 conn->set_quit_callback(
368 [this](const uint64_t &id) {
369 std::unique_lock lock(conns_mtx_);
370 conns_.erase(id);
371 },
372 conn_id);
373
374 {
375 std::unique_lock lock(conns_mtx_);
376 conns_.emplace(conn_id, conn);
377 }
378 start_one(conn).start([](auto &&) {
379 });
380 }
381 }
