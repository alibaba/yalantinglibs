100 [[nodiscard]] std::errc start() noexcept {
101 std::errc ec{};
102 {
103 std::unique_lock lock(start_mtx_);
104 if (flag_ != stat::init) {
105 if (flag_ == stat::started) {
106 easylog::info("start again");
107 }
108 else if (flag_ == stat::started) {
109 easylog::info("has stoped");
110 }
111 return std::errc::io_error;
112 }
113
114 ec = listen();
115 if (ec == err_ok) {
116 thd_ = std::thread([this] {
117 pool_.run();
118 });
119 acceptor_thd_ = std::thread([this] {
120 acceptor_ioc_.run();
121 });
122
123 flag_ = stat::started;
124 }
125 else {
126 flag_ = stat::stop;
127 }
128 }
129
130 cond_.notify_all();
131
132 if (ec == err_ok) {
133 async_simple::coro::syncAwait(accept());
134 }
135 return ec;
136 }

138 [[nodiscard]] bool wait_for_start(auto duration) {
139 std::unique_lock lock(start_mtx_);
140 return cond_.wait_for(lock, duration, [this] {
141 return flag_ == stat::started || flag_ == stat::stop;
142 });

186 void stop() {
187 std::unique_lock lock(start_mtx_);
188 if (flag_ == stat::stop) {
189 return;
190 }
191
192 easylog::info("begin to stop coro_rpc_server, conn size {}", conns_.size());
193 close_acceptor();
194 if (flag_ == stat::started) {
195 {
196 std::unique_lock lock(mtx_);
197 for (auto &conn : conns_) {
198 conn.second->set_quit_callback(nullptr, 0);
199 if (!conn.second->has_closed()) {
200 conn.second->sync_close(false);
201 }
202
203 conn.second->wait_quit();
204 }
205
206 conns_.clear();
207 }
208
209 pool_.stop();
210 }
211 if (thd_.joinable()) {
212 thd_.join();
213 }
214
215 if (acceptor_thd_.joinable()) {
216 acceptor_ioc_.stop();
217 acceptor_thd_.join();
218 }
219 easylog::info("stop coro_rpc_server ok");
220 flag_ = stat::stop;
221 }

236 std::errc listen() {
237 easylog::info("begin to listen");
238 using asio::ip::tcp;
239 auto endpoint = tcp::endpoint(tcp::v4(), port_);
240 acceptor_.open(endpoint.protocol());
241#ifdef __GNUC__
242 acceptor_.set_option(tcp::acceptor::reuse_address(true));
243#endif
244 asio::error_code ec;
245 acceptor_.bind(endpoint, ec);
246 if (ec) {
247 easylog::error("bind port {} error : {}", port_, ec.message());
248 acceptor_.cancel(ec);
249 acceptor_.close(ec);
250 start_accept_promise_.set_value();
251 close_accept_promise_.set_value();
252 return std::errc::address_in_use;
253 }
254#ifdef _MSC_VER
255 acceptor_.set_option(tcp::acceptor::reuse_address(true));
256#endif
257 acceptor_.listen();
258
259 auto end_point = acceptor_.local_endpoint(ec);
260 if (ec) {
261 easylog::error("get local endpoint port {} error {}", port_,
262 ec.message());
263 return std::errc::address_in_use;
264 }
265 port_ = end_point.port();
266
267 easylog::info("listen port {} successfully", port_);
268 start_accept_promise_.set_value();
269 return {};
270 }

272 async_simple::coro::Lazy<std::errc> accept() {
273 for (;;) {
274 auto &io_context = pool_.get_io_context();
275 asio::ip::tcp::socket socket(io_context);
276 auto error = co_await asio_util::async_accept(acceptor_, socket);
277#ifdef UNIT_TEST_INJECT
278 if (g_action == inject_action::force_inject_server_accept_error) {
279 asio::error_code ignored_ec;
280 socket.shutdown(asio::ip::tcp::socket::shutdown_both, ignored_ec);
281 socket.close(ignored_ec);
282 error = make_error_code(std::errc::io_error);
283 // only inject once
284 g_action = inject_action::nothing;
285 }
286#endif
287 if (error) {
288 easylog::error("accept failed, error: {}", error.message());
289 if (error == asio::error::operation_aborted) {
290 close_accept_promise_.set_value();
291 co_return std::errc::io_error;
292 }
293 continue;
294 }
295
296 int64_t conn_id = ++conn_id_;
297 easylog::info("new client conn_id {} coming", conn_id);
298 auto conn = std::make_shared<coro_connection>(
299 io_context, std::move(socket), conn_timeout_duration_);
300 conn->set_quit_callback(
301 [this](const uint64_t &id) {
302 std::unique_lock lock(mtx_);
303 conns_.erase(id);
304 },
305 conn_id);
306
307 {
308 std::unique_lock lock(mtx_);
309 conns_.emplace(conn_id, conn);
310 }
311
312 start_one(conn).via(&executor_).detach();
313 }
314 }

325 void close_acceptor() {
326 start_accept_promise_.get_future().wait();
327 asio::dispatch(acceptor_.get_executor(), [this]() {
328 asio::error_code ec;
329 acceptor_.cancel(ec);
330 acceptor_.close(ec);
331 });
332 close_accept_promise_.get_future().wait();
333 }
