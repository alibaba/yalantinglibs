38class async_rpc_server {
39 enum class stat {
40 init = 0, // server hasn't started/stopped.
41 started, // server is started
42 stop, // server is stopping/stopped
43 };
44
45 public:
53 async_rpc_server(size_t thread_num, unsigned short port,
54 std::chrono::steady_clock::duration conn_timeout_duration =
55 std::chrono::seconds(0))
56 : pool_(thread_num),
57 acceptor_(pool_.get_io_context()),
58 port_(port),
59 conn_timeout_duration_(conn_timeout_duration),
60 signals_(pool_.get_io_context()),
61 flag_(stat::init) {
62 signals_.add(SIGINT);
63 signals_.add(SIGTERM);
64#if defined(SIGQUIT)
65 signals_.add(SIGQUIT);
66#endif // defined(SIGQUIT)
67 signals_.async_wait([this](std::error_code /*ec*/, int /*signo*/) {
68 stop();
69 });
70 }
71#ifdef ENABLE_SSL
72 void init_ssl_context(const ssl_configure &conf) {
73 use_ssl_ = init_ssl_context_helper(context_, conf);
74 }
75#endif

77 [[nodiscard]] std::errc start() {
78 std::errc ec{};
79 {
80 std::unique_lock lock(start_mtx_);
81 if (flag_ != stat::init) {
82 if (flag_ == stat::started) {
83 easylog::info("start again");
84 }
85 else if (flag_ == stat::started) {
86 easylog::info("has stoped");
87 }
88 return std::errc::io_error;
89 }
90
91 ec = init_acceptor();
92 if (ec == err_ok) {
93 do_accept();
94 flag_ = stat::started;
95 }
96 else {
97 flag_ = stat::stop;
98 }
99 }
100
101 cond_.notify_all();
102 if (ec == err_ok) {
103 pool_.run();
104 }
105 return ec;
106 }
107
108 [[nodiscard]] bool wait_for_start(auto duration) {
109 std::unique_lock lock(start_mtx_);
110 return cond_.wait_for(lock, duration, [this] {
111 return flag_ == stat::started || flag_ == stat::stop;
112 });
113 }
114
115 [[nodiscard]] std::errc async_start() {
116 std::errc ec{};
117 {
118 std::unique_lock lock(start_mtx_);
119 if (flag_ != stat::init) {
120 if (flag_ == stat::started) {
121 easylog::info("start again");
122 }
123 else if (flag_ == stat::started) {
124 easylog::info("has stoped");
125 }
126 return std::errc::io_error;
127 }
128
129 std::promise<std::errc> promise;
130 auto future = promise.get_future();
131 thd_ = std::thread([this, &ec, &promise] {
132 ec = init_acceptor();
133 if (ec == std::errc{}) {
134 do_accept();
135 promise.set_value(ec);
136 pool_.run();
137 }
138 else {
139 promise.set_value(ec);
140 }
141 });
142 ec = future.get();
143 flag_ = ec == err_ok ? stat::started : stat::stop;
144 }
145
146 cond_.notify_all();
147
148 return ec;
149 }
150
151 void stop() {
152 std::unique_lock lock(start_mtx_);
153
154 if (flag_ == stat::stop) {
155 if (thd_.joinable()) {
156 thd_.join();
157 }
158 return;
159 }
160 easylog::info("begin to stop async_rpc_server");
161 close_acceptor();
162 if (flag_ == stat::started) {
163 // notify connection quit all async event
164 for (auto &pair : conns_) {
165 if (auto conn = pair.second.lock(); conn) {
166 conn->quit();
167 }
168 }
169 pool_.stop();
170 }
171 if (thd_.joinable()) {
172 thd_.join();
173 }
174
175 easylog::info("stop async_rpc_server ok");
176 flag_ = stat::stop;
177 }
178
179 uint16_t port() const { return port_; };
180
181 ~async_rpc_server() { stop(); }
182
183 private:
184 void do_accept() {
185 auto conn = std::make_shared<async_connection>(pool_.get_io_context(),
186 conn_timeout_duration_);
187#ifdef ENABLE_SSL
188 if (use_ssl_) {
189 conn->init_ssl(context_);
190 }
191#endif
192 if (!promise_) [[unlikely]] {
193 promise_ = std::promise<void>{};
194 }
195 acceptor_.async_accept(
196 conn->socket(), [this, conn](asio::error_code ec) mutable {
197 if (ec) {
198 easylog::info("acceptor error: {}", ec.message());
199 if (ec == asio::error::operation_aborted) {
200 promise_.value().set_value();
201 return;
202 }
203 }
204 else {
205 int64_t conn_id = ++conn_id_;
206 easylog::info("new client conn_id {} coming", conn_id);
207 conns_.emplace(conn_id, conn);
208 conn->start();
209 }
210
211 do_accept();
212 });
213 }
214
215 std::errc init_acceptor() {
216 using asio::ip::tcp;
217 auto endpoint = tcp::endpoint(tcp::v4(), port_);
218 acceptor_.open(endpoint.protocol());
219#ifdef __GNUC__
220 acceptor_.set_option(tcp::acceptor::reuse_address(true));
221#endif
222
223 asio::error_code ec;
224 acceptor_.bind(endpoint, ec);
225 if (ec) {
226 easylog::error("bind port {} error : {}", port_, ec.message());
227 acceptor_.cancel(ec);
228 acceptor_.close(ec);
229 return std::errc::address_in_use;
230 }
231#ifdef _MSC_VER
232 acceptor_.set_option(tcp::acceptor::reuse_address(true));
233#endif
234 acceptor_.listen();
235
236 auto end_point = acceptor_.local_endpoint(ec);
237 if (ec) {
238 easylog::error("get local endpoint port {} error {}", port_,
239 ec.message());
240 return std::errc::address_in_use;
241 }
242 port_ = end_point.port();
243
244 easylog::info("listen port {} successfully", port_);
245 return std::errc{};
246 }
247
248 void close_acceptor() {
249 if (!acceptor_.is_open()) {
250 return;
251 }
252
253 asio::post(acceptor_.get_executor(), [this]() {
254 asio::error_code ec;
255 acceptor_.cancel(ec);
256 acceptor_.close(ec);
257 });
258
259 if (promise_) {
260 promise_.value().get_future().wait();
261 }
262 }
263
264 asio_util::io_context_pool pool_;
265 tcp::acceptor acceptor_;
266 std::atomic<uint16_t> port_;
267 std::chrono::steady_clock::duration conn_timeout_duration_;
268 std::thread thd_;
269 asio::signal_set signals_;
270 std::optional<std::promise<void>> promise_;
271 stat flag_;
272
273 std::mutex start_mtx_;
274 std::condition_variable cond_;
275
276 uint64_t conn_id_ = 0;
277 std::unordered_map<uint64_t, std::weak_ptr<async_connection>> conns_;
278#ifdef ENABLE_SSL
279 asio::ssl::context context_{asio::ssl::context::sslv23};
280 bool use_ssl_ = false;
281#endif
282};
